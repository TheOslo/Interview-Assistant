<script>
        document.addEventListener('DOMContentLoaded', () => {
            const form = document.getElementById('evaluationForm');
            const submitBtn = document.getElementById('submitBtn');
            const btnText = document.getElementById('btnText');
            const btnSpinner = document.getElementById('btnSpinner');
            const inputs = form.querySelectorAll('input, textarea, select');
            const resultArea = document.getElementById('resultArea');

            form.addEventListener('submit', async (e) => {
                e.preventDefault();

                // Disable UI
                inputs.forEach(input => input.disabled = true);
                submitBtn.disabled = true;
                btnText.textContent = 'Evaluating... (This may take up to 25s)';
                btnSpinner.classList.remove('hidden');
                resultArea.classList.add('hidden');
                resultArea.innerHTML = '';

                // Gather data
                const formData = new FormData(form);
                const payload = {
                    problemDescription: formData.get('problemDescription'),
                    programmingLanguage: formData.get('programmingLanguage'),
                    userCode: formData.get('userCode'),
                    skill: formData.get('skill')
                };

                try {
                    // Try the real endpoint, fallback to mock if unreachable (e.g. standard environment)
                    let response;
                    try {
                        response = await fetch('http://localhost:8000/api/evaluate', {
                            method: 'POST',
                            headers: {
                                'Content-Type': 'application/json'
                            },
                            body: JSON.stringify(payload)
                        });
                    } catch (err) {
                        response = await simulateApiCall(payload);
                    }

                    if (!response.ok) {
                        throw new Error(`HTTP error! status: ${response.status}`);
                    }

                    const data = await response.json();
                    
                    // Render Success
                    resultArea.innerHTML = `
                        <div class="result-container result-success">
                            <div class="result-header">
                                <span class="material-symbols-outlined">check_circle</span>
                                <h3>Evaluation Complete</h3>
                            </div>
                            <div class="result-body">${data.evaluation}</div>
                        </div>
                    `;
                    resultArea.classList.remove('hidden');

                } catch (error) {
                    // Render Error
                    resultArea.innerHTML = `
                        <div class="result-container result-error">
                            <div class="result-header">
                                <span class="material-symbols-outlined">error</span>
                                <h3>Evaluation Failed</h3>
                            </div>
                            <div class="result-body" style="color: var(--error)">An error occurred during evaluation. Server might be unreachable or returned an error. (${error.message})</div>
                        </div>
                    `;
                    resultArea.classList.remove('hidden');
                } finally {
                    // Re-enable UI
                    inputs.forEach(input => input.disabled = false);
                    submitBtn.disabled = false;
                    btnText.textContent = 'Evaluate Code';
                    btnSpinner.classList.add('hidden');
                }
            });

            // Mock function to simulate network delay and response
            function simulateApiCall(data) {
                return new Promise((resolve, reject) => {
                    setTimeout(() => {
                        // 80% chance of success for demo purposes
                        if (Math.random() > 0.2) {
                            resolve({
                                ok: true,
                                json: async () => ({
                                    evaluation: "Code structure looks solid. Time complexity is O(n), space complexity is O(1). However, variable naming could be more descriptive. Consider replacing 'arr' with 'inputList'. Overall, a strong approach."
                                })
                            });
                        } else {
                            resolve({
                                ok: false,
                                status: 422
                            });
                        }
                    }, 2000); // 2 second mock delay
                });
            }
        });
    </script>
